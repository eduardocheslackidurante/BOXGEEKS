DROP TABLE IF EXISTS clientes;
DROP TABLE IF EXISTS produtos;
DROP TABLE IF EXISTS funcionario;

-- Tabela de clientes
CREATE TABLE IF NOT EXISTS clientes (
  id_clientes INTEGER PRIMARY KEY AUTOINCREMENT,
  nome VARCHAR(100) NOT NULL,
  cpf_cliente VARCHAR(14) NOT NULL UNIQUE,
  telefone VARCHAR(15),
  email VARCHAR(100)
 );

-- Tabela de produtos
CREATE TABLE IF NOT EXISTS produtos (
  id_produtos INTEGER PRIMARY KEY AUTOINCREMENT,
  nomeprod VARCHAR(100) NOT NULL,
  productPrice DECIMAL(8,2) NOT NULL,
  categoriaprod TEXT,
  productDescription TEXT,
  quantidade_estoque INTEGER
);

CREATE TABLE IF NOT EXISTS funcionario (
  idfunc INTEGER PRIMARY KEY AUTOINCREMENT,
  funcnome VARCHAR(100) NOT NULL,
  cpffunc
  telfunc VARCHAR(15),
  funccargo VARCHAR(30),
  funcdepartamento TEXT,
  funcstatus TEXT
);

SELECT * FROM clientes;
SELECT * FROM produtos;
SELECT * FROM funcionario;
